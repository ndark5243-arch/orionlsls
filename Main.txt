local function AddConnection(Signal, Function)
	if (not OrionLib:IsRunning()) then
		return
	end
	local SignalConnect = Signal:Connect(Function)
	table.insert(OrionLib.Connections, SignalConnect)
	return SignalConnect
end

-- Sistema Universal de Arrastar para PC/Mobile
local function AddDraggingFunctionality(DragPoint, Main)
	pcall(function()
		local Dragging, DragInput, MousePos, FramePos = false
		
		if IsMobile then
			-- Sistema de arrastar para mobile
			local function UpdatePosition(input)
				if Dragging then
					local Delta = input.Position - MousePos
					TweenService:Create(Main, TweenInfo.new(0.1, Enum.EasingStyle.Linear, Enum.EasingDirection.Out), {
						Position = UDim2.new(FramePos.X.Scale, FramePos.X.Offset + Delta.X, FramePos.Y.Scale, FramePos.Y.Offset + Delta.Y)
					}):Play()
				end
			end
			
			AddConnection(DragPoint.InputBegan, function(input)
				if input.UserInputType == Enum.UserInputType.Touch then
					Dragging = true
					MousePos = input.Position
					FramePos = Main.Position
					
					AddConnection(input.Changed, function()
						if input.UserInputState == Enum.UserInputState.End then
							Dragging = false
						end
					end)
				end
			end)
			
			AddConnection(DragPoint.InputChanged, function(input)
				if input.UserInputType == Enum.UserInputType.Touch then
					DragInput = input
				end
			end)
			
			AddConnection(TouchInputService.TouchMoved, function(input, processed)
				if not processed and input == DragInput and Dragging then
					UpdatePosition(input)
				end
			end)
		else
			-- Sistema de arrastar para PC
			AddConnection(DragPoint.InputBegan, function(Input)
				if Input.UserInputType == Enum.UserInputType.MouseButton1 then
					Dragging = true
					MousePos = Input.Position
					FramePos = Main.Position

					Input.Changed:Connect(function()
						if Input.UserInputState == Enum.UserInputState.End then
							Dragging = false
						end
					end)
				end
			end)
			AddConnection(DragPoint.InputChanged, function(Input)
				if Input.UserInputType == Enum.UserInputType.MouseMovement then
					DragInput = Input
				end
			end)
			AddConnection(UserInputService.InputChanged, function(Input)
				if Input == DragInput and Dragging then
					local Delta = Input.Position - MousePos
					TweenService:Create(Main, TweenInfo.new(0.45, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {Position  = UDim2.new(FramePos.X.Scale,FramePos.X.Offset + Delta.X, FramePos.Y.Scale, FramePos.Y.Offset + Delta.Y)}):Play()
				end
			end)
		end
	end)
end

-- Sistema Universal de Input para PC/Mobile
local function AddUniversalInput(Object, Callback, IsHold)
	local InputBegan = Instance.new("BindableEvent")
	local InputEnded = Instance.new("BindableEvent")
	local IsPressed = false
	
	if IsMobile then
		-- Sistema para touch
		AddConnection(Object.InputBegan, function(input)
			if input.UserInputType == Enum.UserInputType.Touch then
				CurrentTouchPositions[input] = Vector2.new(input.Position.X, input.Position.Y)
				ActiveTouchObjects[input] = Object
				IsPressed = true
				InputBegan:Fire()
				
				if IsHold then
					while IsPressed do
						Callback(true)
						wait(0.1)
					end
				else
					Callback()
				end
			end
		end)
		
		AddConnection(Object.InputEnded, function(input)
			if input.UserInputType == Enum.UserInputType.Touch then
				CurrentTouchPositions[input] = nil
				ActiveTouchObjects[input] = nil
				IsPressed = false
				InputEnded:Fire()
			end
		end)
		
		AddConnection(Object.InputChanged, function(input)
			if input.UserInputType == Enum.UserInputType.Touch then
				CurrentTouchPositions[input] = Vector2.new(input.Position.X, input.Position.Y)
			end
		end)
	else
		-- Sistema para mouse
		AddConnection(Object.MouseButton1Down, function()
			IsPressed = true
			InputBegan:Fire()
			
			if IsHold then
				while IsPressed do
					Callback(true)
					wait(0.1)
				end
			else
				Callback()
			end
		end)
		
		AddConnection(Object.MouseButton1Up, function()
			IsPressed = false
			InputEnded:Fire()
		end)
		
		AddConnection(Object.MouseEnter, function()
			if IsHold and IsPressed then
				Callback(true)
			end
		end)
		
		AddConnection(Object.MouseLeave, function()
			if IsHold and IsPressed then
				Callback(false)
			end
		end)
	end
	
	return {
		InputBegan = InputBegan.Event,
		InputEnded = InputEnded.Event
	}
end

-- Otimizar elementos para mobile
local function OptimizeForMobile(object)
	if IsMobile and object then
		-- Aumentar tamanho dos botões para touch
		if object:IsA("TextButton") or object:IsA("ImageButton") or object:IsA("Frame") then
			local padding = 8 -- Padding extra para touch
			if object:FindFirstChild("Content") then
				-- Aumentar fonte em mobile
				local content = object.Content
				if content:IsA("TextLabel") then
					content.TextSize = content.TextSize + 2
				end
			end
		end
		
		-- Aumentar scrollbar thickness
		if object:IsA("ScrollingFrame") then
			object.ScrollBarThickness = 8
		end
	end
end

local function Create(Name, Properties, Children)
	local Object = Instance.new(Name)
	for i, v in next, Properties or {} do
		Object[i] = v
	end
	for i, v in next, Children or {} do
		v.Parent = Object
	end
	
	-- Otimizar para mobile
	OptimizeForMobile(Object)
	
	return Object
end

local function CreateElement(ElementName, ElementFunction)
	OrionLib.Elements[ElementName] = function(...)
		return ElementFunction(...)
	end
end

local function MakeElement(ElementName, ...)
	local NewElement = OrionLib.Elements[ElementName](...)
	return NewElement
end

local function SetProps(Element, Props)
	table.foreach(Props, function(Property, Value)
		Element[Property] = Value
	end)
	return Element
end

local function SetChildren(Element, Children)
	table.foreach(Children, function(_, Child)
		Child.Parent = Element
	end)
	return Element
end

local function Round(Number, Factor)
	local Result = math.floor(Number/Factor + (math.sign(Number) * 0.5)) * Factor
	if Result < 0 then Result = Result + Factor end
	return Result
end

local function ReturnProperty(Object)
	if Object:IsA("Frame") or Object:IsA("TextButton") then
		return "BackgroundColor3"
	end 
	if Object:IsA("ScrollingFrame") then
		return "ScrollBarImageColor3"
	end 
	if Object:IsA("UIStroke") then
		return "Color"
	end 
	if Object:IsA("TextLabel") or Object:IsA("TextBox") then
		return "TextColor3"
	end   
	if Object:IsA("ImageLabel") or Object:IsA("ImageButton") then
		return "ImageColor3"
	end   
end

local function AddThemeObject(Object, Type)
	if not OrionLib.ThemeObjects[Type] then
		OrionLib.ThemeObjects[Type] = {}
	end    
	table.insert(OrionLib.ThemeObjects[Type], Object)
	Object[ReturnProperty(Object)] = OrionLib.Themes[OrionLib.SelectedTheme][Type]
	return Object
end    

local function SetTheme()
	for Name, Type in pairs(OrionLib.ThemeObjects) do
		for _, Object in pairs(Type) do
			Object[ReturnProperty(Object)] = OrionLib.Themes[OrionLib.SelectedTheme][Name]
		end    
	end    
end

local function PackColor(Color)
	return {R = Color.R * 255, G = Color.G * 255, B = Color.B * 255}
end    

local function UnpackColor(Color)
	return Color3.fromRGB(Color.R, Color.G, Color.B)
end

local function LoadCfg(Config)
	local Data = HttpService:JSONDecode(Config)
	table.foreach(Data, function(a,b)
		if OrionLib.Flags[a] then
			spawn(function() 
				if OrionLib.Flags[a].Type == "Colorpicker" then
					OrionLib.Flags[a]:Set(UnpackColor(b))
				else
					OrionLib.Flags[a]:Set(b)
				end    
			end)
		else
			warn("Orion Library Config Loader - Could not find ", a ,b)
		end
	end)
end

local function SaveCfg(Name)
	local Data = {}
	for i,v in pairs(OrionLib.Flags) do
		if v.Save then
			if v.Type == "Colorpicker" then
				Data[i] = PackColor(v.Value)
			else
				Data[i] = v.Value
			end
		end	
	end
	writefile(OrionLib.Folder .. "/" .. Name .. ".txt", tostring(HttpService:JSONEncode(Data)))
end

				OrionLib:MakeNotification({
					Name = "Configuration",
					Content = "Auto-loaded configuration for the game " .. game.GameId .. ".",
					Time = 5
				})
			end
		end)		
	end	
end

			if processed then return end
			
			local currentTime = tick()
			if currentTime - lastTouchTime > 1 then

				touchCount = touchCount + 1
